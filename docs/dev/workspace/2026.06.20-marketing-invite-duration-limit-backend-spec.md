```typescript
const now = new Date();
```

```typescript

if (inviteCode.entitlementDurationDays) {
  currentPeriodEnd = new Date();
  currentPeriodEnd.setDate(
    currentPeriodEnd.getDate() + inviteCode.entitlementDurationDays,
  );
}

// workspaceBillingState insert 수행

  workspaceId: newWorkspaceId,
```

## 5. 혜택 자동 해제 아키텍처 (Entitlement Revocation)

Polar 등 외부 결제 서비스와 달리, 마케팅 코드로 부여받은 권한은 만료 시 Webhook이 날아오지 않으므로 내부 스케줄러에 의한 자동 권한 강등 로직이 반드시 필요합니다.

### 5.1. Cron Service 로직 추가 (`src/domain/billing/services/run-billing-revocation.ts`)

Cloudflare Worker 환경에 맞춰 `CloudflareEnv`를 주입받아 DB를 제어하는 순수 함수를 작성합니다.

```typescript
import { getDb } from "@/db";
import { workspaceBillingState, workspaces } from "@/db/schema";
import { and, eq, inArray, lt } from "drizzle-orm";

export const runBillingRevocation = async (env: CloudflareEnv) => {
  const db = getDb(env.DB);
  
  const expiredStates = await db
    .select()
    .from(workspaceBillingState)
    .where(
      and(
        eq(workspaceBillingState.billingStatus, "ACTIVE"),
        inArray(workspaceBillingState.entitlementSource, [
          "BETA_PROMOTIONAL_GRANT",
          "MANUAL_GRANT",
        ]),
        lt(workspaceBillingState.currentPeriodEnd, new Date()),
      ),
    );

  if (expiredStates.length === 0) return { expiredCount: 0 };

  // D1 성능 최적화를 위해 db.batch() 사용
  const batchStmts = [];
  for (const state of expiredStates) {
    // 1. workspaceBillingState의 billingStatus를 'EXPIRED'로 변경
    batchStmts.push(
      db.update(workspaceBillingState)
        .set({ billingStatus: "EXPIRED" })
        .where(eq(workspaceBillingState.workspaceId, state.workspaceId))
    );
      
    // 2. workspaces 테이블의 planCode를 'FREE'로 강등
    batchStmts.push(
      db.update(workspaces)
        .set({ planCode: "FREE" })
        .where(eq(workspaces.id, state.workspaceId))
    );
  }
  
  await db.batch(batchStmts);
  
  return { expiredCount: expiredStates.length };
};
```

### 5.2. `worker.js` 스케줄러 등록
작성한 `runBillingRevocation` 함수를 `worker.js`의 `scheduled` 이벤트에 연결하여 하루 2번 (KST 자정, UTC 자정) 실행되도록 합니다.

```javascript
import { runBillingRevocation } from "@/domain/billing/services/run-billing-revocation";

// worker.js 내부 scheduled 핸들러
  async scheduled(controller, env, ctx) {
    // 매시간 도는 기본 크론 훅 안에서 현재 시간을 파악
    const currentUTCHour = new Date().getUTCHours();
    
    // UTC 0시 (한국 오전 9시) 또는 UTC 15시 (한국 자정)
    if (currentUTCHour === 0 || currentUTCHour === 15) {
      ctx.waitUntil(
        runBillingRevocation(env)
          .then((res) => console.log("Billing revocation run successfully:", res))
          .catch((err) => console.error("Billing revocation run failed:", err))
      );
    }
    
    // 기존 dailyReminder 등...
  }
```

## 6. 기존 서비스 영향도 검토 (Backward Compatibility)
- **기존 초대코드의 하위 호환성**: 기존 데이터는 `expiresAt`과 `entitlementDurationDays` 필드가 `null`로 조회됩니다. 검사 로직 및 크론 로직 모두 `null`인 경우 "제한 없음(평생 유지)"으로 기존과 완벽하게 동일하게 동작합니다.
- **결제 모델 충돌 여부**: 프로모션 적용 이후 유저가 Polar를 통해 실제 유료 결제를 하면 `entitlementSource`가 `POLAR`로 변경됩니다. 크론 훅은 `BETA_PROMOTIONAL_GRANT` 등 수동 부여 건만 타겟팅하므로 유료 결제자의 권한을 잘못 건드리지 않습니다.
- **운영자 수동 보정**: 운영자가 `MANUAL_GRANT` 또는 `BETA_PROMOTIONAL_GRANT` 권한의 이용 종료일을 현재 시점보다 과거로 저장하면, 스케줄러를 기다리지 않고 즉시 `billingStatus=EXPIRED`, `planCode=FREE`로 정규화합니다. 미래 종료일은 기존처럼 `ACTIVE`로 저장하고 스케줄러가 만료 시점 이후 회수합니다.
- **운영자 전체 동기화**: 기존 데이터가 이미 어긋난 경우를 위해 admin billing 목록 화면에서 `전체 상태 동기화`를 제공한다. 이 버튼은 모든 워크스페이스를 다시 평가하며, 과거 종료일의 `MANUAL_GRANT`/`BETA_PROMOTIONAL_GRANT` `ACTIVE` 또는 `CANCELED` 권한을 `EXPIRED/FREE`로 보정하고 audit log를 남긴다.
